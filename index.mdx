```bash cURL
curl -X POST https://api.ararahq.com/api/v1/messages \
  -H "Authorization: Bearer ara_test_xxx" \
  -H "Content-Type: application/json" \
  -d '{
    "receiver": "whatsapp:+5511999999999",
    "templateName": "hello_world",
    "templateVariables": ["seu_nome"]
  }'
```

```typescript Node.js
import { NodeSDK } from '@ararahq/sdk';

const sdk = new NodeSDK({
  apiKey: process.env.ARARA_API_KEY!,
  baseUrl: 'https://api.ararahq.com/api',
});

await sdk.messages.send({
  receiver: 'whatsapp:+5511999999999',
  templateName: 'hello_world',
  templateVariables: ['seu_nome'],
});
```

```python Python
from arara_api_sdk import AraraClient
from arara_api_sdk.models.message import SendMessageRequest

with AraraClient(api_key="ara_test_xxx") as client:
    client.messages.send(SendMessageRequest(
        receiver="whatsapp:+5511999999999",
        template_name="hello_world",
        template_variables=["seu_nome"],
    ))
```

```php PHP
use Arara\Arara;
use Arara\Config;

$sdk = new Arara(new Config(apiKey: 'ara_test_xxx'));

$sdk->messages->send(
    receiver: 'whatsapp:+5511999999999',
    templateName: 'hello_world',
    variables: ['seu_nome'],
);
```

```java Java
Arara arara = Arara.builder().apiKey("ara_test_xxx").build();

arara.getMessages().send(SendMessageRequest.builder()
    .receiver("whatsapp:+5511999999999")
    .templateName("hello_world")
    .templateVariables(List.of("seu_nome"))
    .build());
```
